TER - Haskell


module Fib where

fib :: Int -> Int
fib 0 = 0
fib 1 = 1
fib n = fib (n - 1) + fib (n - 2)

fib2 :: Integer -> Integer
fib2 n = go n (0, 1) where
  go 0 (a, _) = a
  go n (a, b) = go (n - 1) (b, a + b)

TER - Scala


object Fib {
  def fib(n: Int): Int = n match {
    case 0 => 0
    case 1 => 1
    case _ => fib(n - 1) + fib(n - 2)
  }

  import scala.math.BigInt

  def fib2(n: Int): BigInt = {
    def go(n: Int, acc: (BigInt, BigInt)): BigInt = n match {
      case 0 => acc._1
      case _ => go(n - 1, (acc._2, acc._1 + acc._2))
    }

    go(n, (0, 1))
  }
}

TER - Elixir


defmodule Fib do
  @moduledoc false

  def fib(0), do: 0
  def fib(1), do: 1
  def fib(n), do: fib(n - 1) + fib(n - 2)

  def fib2(n), do: go2(n, {0, 1})
  defp go2(0, {a, _}), do: a
  defp go2(n, {a, b}), do: go2(n - 1, {b, a + b})
end
